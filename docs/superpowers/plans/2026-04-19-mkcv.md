# mkcv Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a web app that maintains CV content in a single YAML file and exports it to Markdown, LaTeX, and PDF via a browser-based split-view editor.

**Architecture:** Three-layer FastAPI app — content layer (Pydantic CVData models), renderer layer (pluggable BaseRenderer implementations), editor layer (vanilla JS frontend with EditorAdapter abstraction). All endpoints validate internally and return unified structured JSON errors. LaTeX export uses Jinja2 templates selectable via a dropdown that is consistent across preview and export.

**Tech Stack:** Python 3.11+, FastAPI, Pydantic v2, PyYAML, Jinja2, subprocess+pdflatex (TeX Live/MiKTeX); Frontend: CodeMirror 5 (CDN), marked.js (CDN); Tests: pytest, pytest-asyncio, httpx

```python
import yaml
from pydantic import ValidationError
from backend.models import CVData


class YAMLParseError(Exception):
    def __init__(self, message: str, details: list[str] | None = None):
        self.error_type = "invalid_yaml"
        self.message = message
        self.details = details or []
        super().__init__(message)


class CVValidationError(Exception):
    def __init__(self, errors: list[str]):
        self.error_type = "validation_error"
        self.message = "CV data failed validation"
        self.errors = errors
        super().__init__(self.message)


def parse_yaml(yaml_str: str) -> CVData:
    try:
        data = yaml.safe_load(yaml_str)
    except yaml.YAMLError as e:
        raise YAMLParseError("Invalid YAML syntax", details=[str(e)])

    if not isinstance(data, dict):
        raise YAMLParseError("YAML must be a mapping at the top level")

    try:
        return CVData(**data)
    except ValidationError as e:
        errors = [
            f"{'.'.join(str(loc) for loc in err['loc'])}: {err['msg']}"
            for err in e.errors()
        ]
        raise CVValidationError(errors=errors)
```

```python
from backend.models import CVData
from backend.renderers.base import BaseRenderer


class MarkdownRenderer(BaseRenderer):
    def render(self, cv: CVData) -> str:
        parts = []

        parts.append(f"# {cv.personal.name}\n")
        contact = [x for x in [cv.personal.email, cv.personal.phone, cv.personal.location] if x]
        if contact:
            parts.append(" · ".join(contact) + "  ")
        links = []
        if cv.personal.linkedin:
            links.append(f"[{cv.personal.linkedin}](https://{cv.personal.linkedin})")
        if cv.personal.github:
            links.append(f"[{cv.personal.github}](https://{cv.personal.github})")
        if cv.personal.website:
            links.append(f"[{cv.personal.website}](https://{cv.personal.website})")
        if links:
            parts.append(" · ".join(links))
        parts.append("")

        if cv.summary:
            parts.append("## Summary\n")
            parts.append(cv.summary.strip())
            parts.append("")

        if cv.experience:
            parts.append("## Work Experience\n")
            for job in cv.experience:
                end = job.end_date or "Present"
                parts.append(f"### {job.title} — {job.company}")
                parts.append(f"*{job.start_date} – {end}*\n")
                for h in job.highlights:
                    parts.append(f"- {h}")
                parts.append("")

        if cv.education:
            parts.append("## Education\n")
            for edu in cv.education:
                gpa = f" · GPA: {edu.gpa}" if edu.gpa else ""
                parts.append(f"### {edu.degree} — {edu.institution}")
                parts.append(f"*{edu.year}*{gpa}\n")

        if cv.skills:
            parts.append("## Skills\n")
            for group in cv.skills:
                parts.append(f"**{group.category}:** {', '.join(group.items)}  ")
            parts.append("")

        if cv.projects:
            parts.append("## Projects\n")
            for proj in cv.projects:
                name_part = f"[{proj.name}](https://{proj.url})" if proj.url else proj.name
                parts.append(f"### {name_part}")
                parts.append(proj.description)
                for h in proj.highlights:
                    parts.append(f"- {h}")
                parts.append("")

        if cv.certifications:
            parts.append("## Certifications\n")
            for cert in cv.certifications:
                issuer = f" — {cert.issuer}" if cert.issuer else ""
                date = f" · {cert.date}" if cert.date else ""
                parts.append(f"**{cert.name}**{issuer}{date}  ")
            parts.append("")

        if cv.publications:
            parts.append("## Publications\n")
            for pub in cv.publications:
                title = f"[{pub.title}](https://{pub.url})" if pub.url else f"**{pub.title}**"
                venue = f" — {pub.venue}" if pub.venue else ""
                date = f" · {pub.date}" if pub.date else ""
                parts.append(f"{title}{venue}{date}  ")
            parts.append("")

        if cv.languages:
            parts.append("## Languages\n")
            parts.append(" · ".join(f"**{l.language}:** {l.proficiency}" for l in cv.languages))
            parts.append("")

        return "\n".join(parts)
```

- [ ] **Step 3: Create backend/templates/classic/cv.tex.j2**

Note: this template uses custom Jinja2 delimiters (`<< >>` for variables, `<% %>` for blocks) to avoid conflicts with LaTeX's `{}` syntax.

```latex
\documentclass[11pt,a4paper]{article}
\usepackage[top=1.5cm,bottom=1.5cm,left=2cm,right=2cm]{geometry}
\usepackage{enumitem}
\usepackage[hidelinks]{hyperref}
\usepackage{titlesec}
\usepackage{microtype}
\setlength{\parindent}{0pt}
\setlength{\parskip}{4pt}

\titleformat{\section}{\large\bfseries\uppercase}{}{0em}{}[{\titlerule[0.8pt]}]
\titlespacing{\section}{0pt}{12pt}{6pt}

\begin{document}

\begin{center}
  {\Huge\bfseries << cv.personal.name >>}\\[6pt]
  <% if cv.personal.email %>\href{mailto:<< cv.personal.email >>}{<< cv.personal.email >>}<% endif %>
  <% if cv.personal.phone %> $\cdot$ << cv.personal.phone >><% endif %>
  <% if cv.personal.location %> $\cdot$ << cv.personal.location >><% endif %>\\[2pt]
  <% if cv.personal.linkedin %>\href{https://<< cv.personal.linkedin >>}{<< cv.personal.linkedin >>}<% endif %>
  <% if cv.personal.github %> $\cdot$ \href{https://<< cv.personal.github >>}{<< cv.personal.github >>}<% endif %>
  <% if cv.personal.website %> $\cdot$ \href{https://<< cv.personal.website >>}{<< cv.personal.website >>}<% endif %>
\end{center}

<% if cv.summary %>
\section{Summary}
<< cv.summary >>
<% endif %>

<% if cv.experience %>
\section{Experience}
<% for job in cv.experience %>
\textbf{<< job.title >>} \hfill << job.start_date >> -- << job.end_date if job.end_date else "Present" >>\\
\textit{<< job.company >>}
<% if job.highlights %>
\begin{itemize}[leftmargin=*,noitemsep,topsep=2pt]
<% for item in job.highlights %>
  \item << item >>
<% endfor %>
\end{itemize}
<% endif %>
<% if not loop.last %>\vspace{4pt}<% endif %>
<% endfor %>
<% endif %>

<% if cv.education %>
\section{Education}
<% for edu in cv.education %>
\textbf{<< edu.degree >>} \hfill << edu.year >>\\
\textit{<< edu.institution >>}<% if edu.gpa %> \hfill GPA: << edu.gpa >><% endif %>
<% endfor %>
<% endif %>

<% if cv.skills %>
\section{Skills}
<% for group in cv.skills %>
\textbf{<< group.category >>:} << group.items | join(", ") >>\\
<% endfor %>
<% endif %>

<% if cv.projects %>
\section{Projects}
<% for proj in cv.projects %>
\textbf{<% if proj.url %>\href{https://<< proj.url >>}{<< proj.name >>}<% else %><< proj.name >><% endif %>} --- << proj.description >>
<% if proj.highlights %>
\begin{itemize}[leftmargin=*,noitemsep,topsep=2pt]
<% for item in proj.highlights %>
  \item << item >>
<% endfor %>
\end{itemize}
<% endif %>
<% endfor %>
<% endif %>

<% if cv.certifications %>
\section{Certifications}
<% for cert in cv.certifications %>
\textbf{<< cert.name >>}<% if cert.issuer %> --- << cert.issuer >><% endif %><% if cert.date %> \hfill << cert.date >><% endif %>\\
<% endfor %>
<% endif %>

<% if cv.publications %>
```

```python
from pathlib import Path
import jinja2
from backend.models import CVData
from backend.renderers.base import BaseRenderer


class LaTeXRenderer(BaseRenderer):
    def __init__(self, templates_dir: Path, template: str = "classic"):
        self.templates_dir = templates_dir
        self.template = template

    def render(self, cv: CVData) -> str:
        template_path = self.templates_dir / self.template / "cv.tex.j2"
        if not template_path.exists():
            raise ValueError(f"unknown_template: '{self.template}' not found")

        env = jinja2.Environment(
            loader=jinja2.FileSystemLoader(str(self.templates_dir / self.template)),
            block_start_string="<%",
            block_end_string="%>",
            variable_start_string="<<",
            variable_end_string=">>",
            comment_start_string="<#",
            comment_end_string="#>",
            trim_blocks=True,
            lstrip_blocks=True,
        )
        return env.get_template("cv.tex.j2").render(cv=cv)
```

```python
@app.post("/api/validate")
async def validate(req: CVRequest):
    errors: list[str] = []
    try:
        parse_yaml(req.yaml)

    except CVValidationError as e:
        errors.extend(e.errors)

async def export_pdf(req: CVRequest):
    try:
        cv = parse_yaml(req.yaml)
    except YAMLParseError as e:
        return _error("invalid_yaml", e.message, e.details)
    except CVValidationError as e:
        return _error("validation_error", e.message, e.errors)

    if not _template_exists(req.template):
        return _error("unknown_template", f"Template '{req.template}' not found")

    latex_content = LaTeXRenderer(TEMPLATES_DIR, template=req.template).render(cv)

    with tempfile.TemporaryDirectory() as tmpdir:
        tex_path = Path(tmpdir) / "cv.tex"
        tex_path.write_text(latex_content)
        try:
            result = subprocess.run(
                ["pdflatex", "-interaction=nonstopmode", "cv.tex"],
                cwd=tmpdir,
                capture_output=True,
                timeout=30,
                text=True,
            )
        except subprocess.TimeoutExpired:
            return _error("pdf_generation_failed", "pdflatex timed out after 30 seconds")
        except FileNotFoundError:
            return _error("pdf_generation_failed", "pdflatex not found — install TeX Live or MiKTeX")

        if result.returncode != 0:
            details = [line for line in result.stderr.splitlines() if line.strip()]
            return _error("pdf_generation_failed", "pdflatex exited with errors", details)

        pdf_bytes = (Path(tmpdir) / "cv.pdf").read_bytes()

    OUTPUT_DIR.mkdir(exist_ok=True)
    (OUTPUT_DIR / "cv.pdf").write_bytes(pdf_bytes)
    return Response(
        content=pdf_bytes,
        media_type="application/pdf",
        headers={"Content-Disposition": "attachment; filename=cv.pdf"},
    )

    return {"templates": templates}
```

## Adding a Form Editor Later

1. Create `frontend/form-adapter.js` implementing `getValue() → string` and `onChange(callback)`
2. In `frontend/editor-adapter.js`, replace `new CodeMirrorAdapter(...)` with `new FormAdapter(...)` in the DOMContentLoaded handler
3. No other files change
